# layout.py — C43 ASCII Layout Engine Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `layout.py`, a single-file stdlib-only script that turns a `layout.json` (nodes on a coarse grid + edges + optional hints) into a rendered ASCII diagram (`result.txt`) plus a fully-resolved `result.json`, then drive it via the c43 skill until the rebob diagram is visually approved.

**Architecture:** A five-stage pipeline — parse/validate → geometry → port assignment → routing (cell-grid A* restricted to lane bands, two passes) → render/report. A single `Canvas` (2D char grid) is mutated and re-saved to `result.txt` after each logical mutation (scaffolding, each box, each edge) so a crash mid-run still shows progress. `result.json` is the input fully resolved (ports positioned, routes as polylines) plus status/errors.

**Tech Stack:** Python 3 (stdlib only), `uv run` for execution, `pytest` (via `uv run --with pytest pytest`) for tests.

All code lives in one file: `.claude/skills/c43/scripts/layout.py`. Define these at the top once (Task 1). Later tasks add functions that consume/produce them. **Do not rename these between tasks.**

```python
import json, sys, string, heapq
from dataclasses import dataclass, field

EDGE_ALPHABET = string.digits + string.ascii_lowercase + string.ascii_uppercase  # 62 chars
SIDES = ("left", "right", "top", "bottom")

# Geometry constants (tunable during the visual-approval loop)
GUTTER_W   = 8     # label gutter width; the spine '│' sits at column index GUTTER_W
LABEL_PAD  = 4     # total horizontal padding added to widest label -> box width
BOX_H      = 11    # fixed box height in characters
BOX_MARGIN_X = 4   # chars between a node column's left edge and its box
BOX_MARGIN_Y = 2   # chars between a node row's top edge and its box
LANE_MIN_W = 16    # minimum width (chars) of a vertical edge lane
LANE_MIN_H = 7     # minimum height (chars) of a horizontal edge lane
TITLE_H    = 6     # height of the title region (row 000)

class ValidationError(Exception):
    """Stage-1 hard error: no result.txt is produced, only result.json."""

@dataclass
class Node:
    id: str
    label: str
    grid_col: int
    grid_row: int
    x: int = 0; y: int = 0; w: int = 0; h: int = 0   # filled by geometry

@dataclass
class Port:
    side: str
    x: int
    y: int

@dataclass
class Edge:
    id: str
    from_id: str
    to_id: str
    char: str
    from_port: Port = None
    to_port: Port = None
    route: list = None          # list of [x, y] polyline vertices

@dataclass
class LayoutError:
    code: str                   # "validation" | "crossing" | "unroutable"
    edge_ids: list
    at: list                    # [x, y] or None
    message: str
    suggestion: str

@dataclass
class Model:
    title: str
    description: str
    nodes: list                 # list[Node]
    edges: list                 # list[Edge]
    hint_ports: dict = field(default_factory=dict)   # edge_id -> {"from_side","to_side"}
    routing_order: list = field(default_factory=list)
    canvas_w: int = 0
    canvas_h: int = 0
    box_w: int = 0
    box_h: int = 0
    errors: list = field(default_factory=list)        # list[LayoutError]
```

### Canvas coordinate model

- Character canvas is a grid of `(x, y)` cells, `x` = column, `y` = row, origin top-left. `result.txt` and `result.json` share these coordinates.
- **Gutter:** columns `0..GUTTER_W-1` hold row labels; the spine `│` sits at column `GUTTER_W`. Drawable content begins at `GUTTER_W + 1`.
- **Canvas grid columns** (left→right): for `M = max(grid_col)`, region `2c` = node column `c`, region `2c+1` = vertical lane after node column `c` (`c = 0..M`). Total `2*(M+1)` regions.
- **Canvas grid rows** (top→bottom): region `0` = title; region `2r+1` = node row `r`; region `2r+2` = horizontal lane after node row `r` (`r = 0..R`, `R = max(grid_row)`).
- Node columns all share width `box_w + 2*BOX_MARGIN_X`. Node rows all share height `BOX_H + 2*BOX_MARGIN_Y`. Lanes start at `LANE_MIN_W` / `LANE_MIN_H` and grow if routing needs more parallel tracks (so "lane capacity exceeded" is never an error).
- A **track** is one specific character column inside a vertical lane band (or one row inside a horizontal lane band). Cell-grid A* over lane-band cells naturally assigns each edge distinct tracks because painted/claimed cells are marked occupied.

```python
def parse_and_validate(raw):
    nodes_raw = raw.get("nodes", [])
    edges_raw = raw.get("edges", [])

    seen_ids, cells = set(), {}
    nodes = []
    for nr in nodes_raw:
        nid = nr["id"]
        if nid in seen_ids:
            raise ValidationError(f"duplicate node id: {nid}")
        seen_ids.add(nid)
        cell = (nr["grid_col"], nr["grid_row"])
        if cell in cells:
            raise ValidationError(
                f"two nodes in grid cell {cell}: {cells[cell]} and {nid}")
        cells[cell] = nid
        nodes.append(Node(id=nid, label=nr["label"],
                          grid_col=nr["grid_col"], grid_row=nr["grid_row"]))

    if len(edges_raw) > len(EDGE_ALPHABET):
        raise ValidationError(
            f"{len(edges_raw)} edges exceeds the {len(EDGE_ALPHABET)}-char "
            f"alphabet; unicode edge alphabet is the future expansion path")

    node_ids = {n.id for n in nodes}
    seen_edge_ids = set()
    edges = []
    for i, er in enumerate(edges_raw):
        eid = er["id"]
        if eid in seen_edge_ids:
            raise ValidationError(f"duplicate edge id: {eid}")
        seen_edge_ids.add(eid)
        for end in ("from", "to"):
            if er[end] not in node_ids:
                raise ValidationError(
                    f"edge {eid} references unknown node id: {er[end]}")
        edges.append(Edge(id=eid, from_id=er["from"], to_id=er["to"],
                          char=EDGE_ALPHABET[i]))

    hints = raw.get("hints") or {}
    hint_ports = {}
    for hp in hints.get("ports", []):
        if hp["edge_id"] not in seen_edge_ids:
            raise ValidationError(f"hint references unknown edge_id: {hp['edge_id']}")
        for key in ("from_side", "to_side"):
            if key in hp and hp[key] not in SIDES:
                raise ValidationError(f"invalid side name in hint: {hp[key]}")
        hint_ports[hp["edge_id"]] = hp

    routing_order = hints.get("routing_order", [])
    for eid in routing_order:
        if eid not in seen_edge_ids:
            raise ValidationError(f"routing_order references unknown edge_id: {eid}")

    return Model(title=raw.get("title", ""), description=raw.get("description", ""),
                 nodes=nodes, edges=edges,
                 hint_ports=hint_ports, routing_order=routing_order)
```

Add to `layout.py`. It computes region offsets, then places each node centered in its node-column region. Vertical lanes start at `LANE_MIN_W`, horizontal lanes at `LANE_MIN_H` (routing may grow them later; geometry seeds the minimum).

```python
def geometry(m):
    max_label = max((len(n.label) for n in m.nodes), default=0)
    m.box_w = max_label + LABEL_PAD
    m.box_h = BOX_H

    M = max((n.grid_col for n in m.nodes), default=0)
    R = max((n.grid_row for n in m.nodes), default=0)

    node_col_w = m.box_w + 2 * BOX_MARGIN_X
    node_row_h = m.box_h + 2 * BOX_MARGIN_Y

    # x offset of the left edge of each canvas grid column region
    # region 0 begins just right of the spine (GUTTER_W) + 1 separator space
    m.col_x = {}            # region index -> left char column
    x = GUTTER_W + 1
    for c in range(M + 1):
        m.col_x[2 * c] = x          # node column c
        x += node_col_w
        m.col_x[2 * c + 1] = x      # vertical lane c
        x += LANE_MIN_W
    m.canvas_w = x

    # y offset of the top edge of each canvas grid row region
    m.row_y = {}            # region index -> top char row
    y = 0
    m.row_y[0] = y                  # title
    y += TITLE_H
    for r in range(R + 1):
        m.row_y[2 * r + 1] = y      # node row r
        y += node_row_h
        m.row_y[2 * r + 2] = y      # horizontal lane r
        y += LANE_MIN_H
    m.canvas_h = y

    for n in m.nodes:
        n.w = m.box_w
        n.h = m.box_h
        n.x = m.col_x[2 * n.grid_col] + BOX_MARGIN_X
        n.y = m.row_y[2 * n.grid_row + 1] + BOX_MARGIN_Y
```

```python
def _default_sides(src, dst):
    if dst.grid_col > src.grid_col:
        return "right", "left"
    if dst.grid_col < src.grid_col:
        return "left", "right"
    # same column
    if dst.grid_row > src.grid_row:
        return "bottom", "top"
    return "top", "bottom"

def _other_endpoint(m, node, edge):
    other_id = edge.to_id if edge.from_id == node.id else edge.from_id
    return next(n for n in m.nodes if n.id == other_id)

def assign_ports(m):
    by_id = {n.id: n for n in m.nodes}

    # 1. decide a side for each (edge, endpoint)
    plan = {}   # (edge_id, "from"/"to") -> side
    for e in m.edges:
        src, dst = by_id[e.from_id], by_id[e.to_id]
        fs, ts = _default_sides(src, dst)
        h = m.hint_ports.get(e.id)
        if h:
            fs = h.get("from_side", fs)
            ts = h.get("to_side", ts)
        plan[(e.id, "from")] = fs
        plan[(e.id, "to")] = ts

    # 2. group ports by (node, side), order them, stack on distinct cells
    groups = {}   # (node_id, side) -> list[(edge, end)]
    for e in m.edges:
        groups.setdefault((e.from_id, plan[(e.id, "from")]), []).append((e, "from"))
        groups.setdefault((e.to_id, plan[(e.id, "to")]), []).append((e, "to"))

    for (node_id, side), members in groups.items():
        node = by_id[node_id]
        # order by the other endpoint's center to reduce immediate crossings
        def keyfn(item):
            e, _end = item
            other = _other_endpoint(m, node, e)
            return (other.y + other.h / 2) if side in ("left", "right") else (other.x + other.w / 2)
        members.sort(key=keyfn)

        n = len(members)
        for i, (e, end) in enumerate(members):
            if side in ("left", "right"):
                x = node.x if side == "left" else node.x + node.w - 1
                # spread across interior rows, leaving the corners alone
                y = node.y + 1 + (i + 1) * (node.h - 2) // (n + 1)
                port = Port(side=side, x=x, y=y)
            else:
                y = node.y if side == "top" else node.y + node.h - 1
                x = node.x + 1 + (i + 1) * (node.w - 2) // (n + 1)
                port = Port(side=side, x=x, y=y)
            if end == "from":
                e.from_port = port
            else:
                e.to_port = port

### Task 5: Stage 4 — routing (cell-grid A*, two passes, tracks, crossings)

**Files:**
- Modify: `.claude/skills/c43/scripts/layout.py`
- Test: `.claude/skills/c43/scripts/tests/test_routing.py`

Routing is cell-grid A* restricted to **routable cells** (lane bands + the single cell just outside each port). Boxes (interior + border, except the port cell itself) and the gutter are never routable. Cost is lexicographic `(turns, length)`: turning costs `(1, 1)` per step, going straight costs `(0, 1)`. Each routed edge marks the cells it uses as occupied → the next edge picks a different track. **Pass 1** blocks cells already occupied (no crossings). **Pass 2** re-routes only failed edges, allowing crossing an occupied perpendicular cell at high cost; each such crossing is a `crossing` error. Edges still unroutable become `unroutable` errors and are dropped. Edge order: `routing_order` first, then remaining by ascending Manhattan distance.

Lanes grow implicitly: the routable region between two node columns is the full `LANE_MIN_W` band; the A* picks distinct columns (tracks) within it. If more edges need the band than its width, the band must widen — implement `_ensure_lane_capacity(m)` that, before routing, counts edges that will traverse each lane and widens that lane's region (shifting everything to its right) so `band_width >= tracks_needed`. Keep it simple: widen every vertical lane to `max(LANE_MIN_W, vcount)` and every horizontal lane to `max(LANE_MIN_H, hcount)` where counts are upper bounds (number of edges whose port columns/rows straddle that lane). Recompute `col_x`/`row_y`/node coords after widening.

The route polyline stored on each edge is `[from_port_exit, ...bends..., to_port_entry]` — the cell just outside the source port, every bend, the cell just outside the target port. Endpoints in `result.json` `route` are the cells adjacent to the ports (the ports themselves are `from_port`/`to_port`).

def _build_blocked(m):
    """Return a set of (x,y) cells that are never routable (boxes + gutter)."""
    blocked = set()
    for x in range(m.canvas_w):
        for y in range(m.canvas_h):
            if x <= GUTTER_W:
                blocked.add((x, y))
    for n in m.nodes:
        for x in range(n.x, n.x + n.w):
            for y in range(n.y, n.y + n.h):
                blocked.add((x, y))
    return blocked

def _port_exit(port):
    """Cell just outside the box adjacent to the port."""
    if port.side == "left":   return (port.x - 1, port.y)
    if port.side == "right":  return (port.x + 1, port.y)
    if port.side == "top":    return (port.x, port.y - 1)
    return (port.x, port.y + 1)

def _astar(start, goal, passable, occupied, allow_cross):
    """Lexicographic (turns, length) A* on the 4-connected char grid.
    passable(x,y) -> bool for the base mask. occupied is a set of claimed cells;
    blocked unless allow_cross (then entering one adds a high penalty).
    Returns (path_cells, crossings) or (None, None)."""
    CROSS_PENALTY = 10_000
    # state includes direction to count turns
    def h(p):  # manhattan, admissible for length; turns ignored in heuristic
        return abs(p[0] - goal[0]) + abs(p[1] - goal[1])
    start_state = (start, None)
    # priority: (turns, length, crossings) lexicographic via tuple
    pq = [((0, 0, 0), start_state)]
    best = {start_state: (0, 0, 0)}
    came = {}
    while pq:
        cost, (pos, dirn) = heapq.heappop(pq)
        if pos == goal:
            # reconstruct
            cells = [pos]; st = (pos, dirn); crossings = []
            while st in came:
                st = came[st]; cells.append(st[0])
            cells.reverse()
            for c in cells:
                if c in occupied and c != start and c != goal:
                    crossings.append(c)
            return cells, crossings
        if cost > best.get((pos, dirn), (float("inf"),)*3):
            continue
        for d in ((1, 0), (-1, 0), (0, 1), (0, -1)):
            nx, ny = pos[0] + d[0], pos[1] + d[1]
            np = (nx, ny)
            if not (0 <= nx < m_canvas_w_GLOBAL[0] and 0 <= ny < m_canvas_h_GLOBAL[0]):
                continue
            if np != goal and not passable(nx, ny):
                continue
            crossed = np in occupied and np != goal
            if crossed and not allow_cross:
                continue
            turns, length, crs = cost
            nturns = turns + (1 if dirn is not None and d != dirn else 0)
            ncost = (nturns, length + 1, crs + (CROSS_PENALTY if crossed else 0))
            nstate = (np, d)
            if ncost < best.get(nstate, (float("inf"),)*3):
                best[nstate] = ncost
                came[nstate] = (pos, dirn)
                heapq.heappush(pq, (ncost, nstate))
    return None, None

# A* needs canvas bounds; pass them via a tiny module-level box to avoid
# threading through every recursive call. Set in route_all before use.
m_canvas_w_GLOBAL = [0]
m_canvas_h_GLOBAL = [0]

def _manhattan(e):
    return abs(e.from_port.x - e.to_port.x) + abs(e.from_port.y - e.to_port.y)

def route_all(m):
    m_canvas_w_GLOBAL[0] = m.canvas_w
    m_canvas_h_GLOBAL[0] = m.canvas_h
    base_blocked = _build_blocked(m)
    occupied = set()                 # cells claimed by already-routed edges

    def passable(x, y):
        return (x, y) not in base_blocked

    # ordering: routing_order first (in given order), then shortest-manhattan
    order_index = {eid: i for i, eid in enumerate(m.routing_order)}
    ordered = sorted(m.edges,
                     key=lambda e: (order_index.get(e.id, len(order_index)), _manhattan(e)))

    def route_one(e, allow_cross):
        start = _port_exit(e.from_port)
        goal = _port_exit(e.to_port)
        cells, crossings = _astar(start, goal, passable, occupied, allow_cross)
        if cells is None:
            return None, None
        return cells, crossings

    def claim(cells):
        for c in cells:
            occupied.add(c)

    def to_polyline(cells):
        # collapse collinear runs into vertices (bends only)
        poly = [list(cells[0])]
        for i in range(1, len(cells) - 1):
            a, b, c = cells[i - 1], cells[i], cells[i + 1]
            if not ((a[0] == b[0] == c[0]) or (a[1] == b[1] == c[1])):
                poly.append(list(b))
        poly.append(list(cells[-1]))
        return poly

    failed = []
    # Pass 1 — no crossings
    for e in ordered:
        cells, _ = route_one(e, allow_cross=False)
        if cells is None:
            failed.append(e)
            continue
        claim(cells)
        e.route = to_polyline(cells)

    # Pass 2 — desperation, crossings allowed at high cost
    for e in failed:
        cells, crossings = route_one(e, allow_cross=True)
        if cells is None:
            m.errors.append(LayoutError(
                code="unroutable", edge_ids=[e.id], at=None,
                message=f"edge {e.id} could not be routed even with crossings",
                suggestion="move its endpoints to adjacent grid cells, or free a "
                           "lane via hints.routing_order"))
            continue
        claim(cells)
        e.route = to_polyline(cells)
        for c in crossings:
            m.errors.append(LayoutError(
                code="crossing", edge_ids=[e.id], at=list(c),
                message=f"edge {e.id} crosses another edge at {c}",
                suggestion="reorder with hints.routing_order or move a node to "
                           "open a parallel track"))
```

class Canvas:
    def __init__(self, w, h):
        self.w, self.h = w, h
        self.grid = [[" "] * w for _ in range(h)]

    def paint(self, x, y, ch):
        if 0 <= x < self.w and 0 <= y < self.h:
            self.grid[y][x] = ch

    def char_at(self, x, y):
        return self.grid[y][x]

    def __str__(self):
        return "\n".join("".join(row).rstrip() for row in self.grid) + "\n"

    def save(self, path):
        with open(path, "w") as f:
            f.write(str(self))

ARROW = {"left": "◄", "right": "►", "top": "▲", "bottom": "▼"}

def _paint_scaffolding(m, cv):
    spine = GUTTER_W
    # vertical spine + lane separators
    for y in range(m.canvas_h):
        cv.paint(spine, y, "│")
    # column header labels (000 nodes / 001 edges ...) in the title region
    # region 2c -> "%03d" + ("nodes" or "edges")
    region = 0
    x = GUTTER_W + 1
    # walk regions in canvas order using col_x
    ordered_regions = sorted(m.col_x.items(), key=lambda kv: kv[1])
    for ridx, rx in ordered_regions:
        kind = "nodes" if ridx % 2 == 0 else "edges"
        for i, ch in enumerate(f"{ridx:03d}"):
            cv.paint(rx + 1 + i, 1, ch)
        for i, ch in enumerate(kind):
            cv.paint(rx + 1 + i, 2, ch)
        # vertical separator at the left of each region (except first)
        if rx > GUTTER_W + 1:
            for y in range(m.canvas_h):
                if cv.char_at(rx - 1, y) == " ":
                    cv.paint(rx - 1, y, "│")
    # row gutter labels + horizontal separators
    for ridx, ry in sorted(m.row_y.items(), key=lambda kv: kv[1]):
        kind = "title" if ridx == 0 else ("nodes" if ridx % 2 == 1 else "edges")
        for i, ch in enumerate(f"{ridx:03d}"):
            cv.paint(2 + i, ry + 1, ch)
        for i, ch in enumerate(kind):
            cv.paint(2 + i, ry + 2, ch)
        if ry > 0:
            for x in range(GUTTER_W):
                cv.paint(x, ry, "─" if x else " ")
            cv.paint(GUTTER_W, ry, "┼")
            for x in range(GUTTER_W + 1, m.canvas_w):
                cv.paint(x, ry, "─")

def _draw_box(cv, n):
    x0, y0, x1, y1 = n.x, n.y, n.x + n.w - 1, n.y + n.h - 1
    for x in range(x0, x1 + 1):
        cv.paint(x, y0, "-"); cv.paint(x, y1, "-")
    for y in range(y0, y1 + 1):
        cv.paint(x0, y, "|"); cv.paint(x1, y, "|")
    for cx, cy in ((x0, y0), (x1, y0), (x0, y1), (x1, y1)):
        cv.paint(cx, cy, "+")
    lx = x0 + (n.w - len(n.label)) // 2
    ly = y0 + n.h // 2
    for i, ch in enumerate(n.label):
        cv.paint(lx + i, ly, ch)

def _paint_edge(cv, e):
    if e.route is None:
        return
    # body: walk polyline filling every cell with the edge char
    pts = e.route
    for (x0, y0), (x1, y1) in zip(pts, pts[1:]):
        if x0 == x1:
            for y in range(min(y0, y1), max(y0, y1) + 1):
                cv.paint(x0, y, e.char)
        else:
            for x in range(min(x0, x1), max(x0, x1) + 1):
                cv.paint(x, y0, e.char)
    # ports
    cv.paint(e.from_port.x, e.from_port.y, "*")
    # arrowhead at the target port, pointing into the box
    cv.paint(e.to_port.x, e.to_port.y, ARROW[_arrow_dir(e.to_port.side)])

def _arrow_dir(to_side):
    # arrow points INTO the box: a left-side port is entered moving right
    return {"left": "right", "right": "left", "top": "bottom", "bottom": "top"}[to_side]

def render(m, cv, path="result.txt"):
    _paint_scaffolding(m, cv); cv.save(path)
    for n in m.nodes:
        _draw_box(cv, n); cv.save(path)
    for e in m.edges:
        if e.route is not None:
            _paint_edge(cv, e); cv.save(path)

    return None if p is None else {"side": p.side, "x": p.x, "y": p.y}

def _result_json(m):
    return {
        "status": "error" if m.errors else "ok",
        "errors": [{"code": e.code, "edge_ids": e.edge_ids, "at": e.at,
                    "message": e.message, "suggestion": e.suggestion}
                   for e in m.errors],
        "title": m.title, "description": m.description,
        "canvas": {"width": m.canvas_w, "height": m.canvas_h},
        "box": {"width": m.box_w, "height": m.box_h},
        "nodes": [{"id": n.id, "label": n.label,
                   "grid_col": n.grid_col, "grid_row": n.grid_row,
                   "x": n.x, "y": n.y, "w": n.w, "h": n.h} for n in m.nodes],
        "edges": [{"id": e.id, "from": e.from_id, "to": e.to_id, "char": e.char,
                   "from_port": _port_json(e.from_port),
                   "to_port": _port_json(e.to_port),
                   "route": e.route} for e in m.edges],
    }

def main(argv):
    if len(argv) < 2:
        print("usage: layout.py layout.json", file=sys.stderr)
        sys.exit(2)
    with open(argv[1]) as f:
        raw = json.load(f)

    try:
        m = parse_and_validate(raw)
    except ValidationError as exc:
        err = {"status": "error",
               "errors": [{"code": "validation", "edge_ids": [], "at": None,
                           "message": str(exc),
                           "suggestion": "fix layout.json per the message above"}],
               "title": raw.get("title", ""), "description": raw.get("description", "")}
        with open("result.json", "w") as f:
            json.dump(err, f, indent=2)
        return

    geometry(m)
    assign_ports(m)
    route_all(m)
    cv = Canvas(m.canvas_w, m.canvas_h)
    render(m, cv, "result.txt")
    with open("result.json", "w") as f:
        json.dump(_result_json(m), f, indent=2)
